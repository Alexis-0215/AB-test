import pandas as pd
import numpy as np

import seaborn as sns

# Global Configuration
sns.set(style="whitegrid", palette="muted")
plt.rcParams['figure.dpi'] = 150
pd.set_option('display.precision', 3)
DATA_PATH = 'ab_data.csv'
ALPHA = 0.05
MIN_EFFECT_SIZE = 0.01  # Minimum business meaningful effect size

def parse_custom_timestamp(time_str: str) -> pd.Timedelta:
    """Parse hybrid format time strings (supports 'HH:MM' and 'HH:MM.SS' formats)"""
    if re.match(r"^\d+:\d+\.?\d*$", time_str):  # Handle "11:48.6" format
        hours_part, minutes_part = time_str.split(':')
        hours = int(hours_part)
        minutes = float(minutes_part)
    elif re.match(r"^\d+\.\d+$", time_str):  # Handle float format like 12.34
        time_float = float(time_str)
        hours = int(time_float)
        minutes = (time_float - hours) * 100
    else:
        raise ValueError(f"Unrecognized time format: {time_str}")
    
    # Validate minute component
    if minutes >= 60:
        hours += 1
        minutes -= 60
        
    return pd.Timedelta(
        hours=hours,
        minutes=int(minutes),
        seconds=round((minutes - int(minutes)) * 60)
    )

def load_and_validate_data(filepath: str) -> pd.DataFrame:
    """Load and validate dataset with comprehensive checks"""
    try:
        df = pd.read_csv(filepath)
        
        # Schema validation
        required_columns = {'user_id', 'timestamp', 'group', 'landing_page', 'converted'}
        if not required_columns.issubset(df.columns):
            missing = required_columns - set(df.columns)
            raise ValueError(f"Missing required columns: {missing}")
            
        # Time format conversion
        print(" Converting timestamps...")
        df['time_offset'] = df['timestamp'].apply(parse_custom_timestamp)
        df = df.sort_values('time_offset')
        
        print(" Data loaded successfully")
        print(f" Total experiment duration: {df['time_offset'].max() - df['time_offset'].min()}")
        return df
    
    except FileNotFoundError:
        raise SystemExit(" Error: Data file not found")

def perform_eda(df: pd.DataFrame) -> None:
    """Execute comprehensive exploratory data analysis"""
    print("\n Performing exploratory analysis...")
    
    # Create time-based features
    df['total_minutes'] = df['time_offset'].dt.total_seconds() / 60
    df['hour'] = df['time_offset'].dt.components['hours']
    df['minute'] = df['time_offset'].dt.components['minutes']
    
    # Initialize visualization framework
    fig, ax = plt.subplots(2, 2, figsize=(14, 10))
    plt.suptitle('Experimental Data Analysis', y=1.02)
    
    # Temporal distribution analysis
    sns.kdeplot(data=df, x='total_minutes', hue='group', ax=ax[0,0])
    ax[0,0].set_title('Temporal Distribution by Group')
    ax[0,0].set_xlabel('Minutes from Start')
    
    # Conversion rate by hour
    hourly_stats = df.groupby(['hour', 'group'])['converted'].mean().reset_index()
    sns.lineplot(data=hourly_stats, x='hour', y='converted', hue='group', ax=ax[0,1])
    ax[0,1].set_title('Conversion Rate by Hour')
    ax[0,1].set_ylim(0, 0.2)
    
    # Group size comparison
    group_counts = df['group'].value_counts()
    ax[1,0].pie(group_counts, labels=group_counts.index, autopct='%1.1f%%')
    ax[1,0].set_title('Group Allocation Distribution')
    
    # Raw conversion rates
    sns.barplot(data=df, x='group', y='converted', ax=ax[1,1], errorbar=None)
    ax[1,1].set_title('Raw Conversion Rates')
    ax[1,1].set_ylim(0, 0.15)
    
    plt.tight_layout()
    plt.show()

def preprocess_data(df: pd.DataFrame) -> pd.DataFrame:
    """
    Clean and validate experimental data
    
    1. Remove duplicate user records
    2. Validate group assignments
    3. Filter invalid test participants
    """
    original_size = df.shape[0]
    
    # Deduplication
    df = df.drop_duplicates(subset=['user_id'], keep='first')
    
    # Validate experimental assignments
    valid_control = (df['group'] == 'control') & (df['landing_page'] == 'old_page')
    valid_treatment = (df['group'] == 'treatment') & (df['landing_page'] == 'new_page')
    valid_records = valid_control | valid_treatment
    
    df = df[valid_records].copy()
    df.reset_index(drop=True, inplace=True)
    
    # Data quality reporting
    print(f"\n Data Quality Report:")
    print(f" - Removed {original_size - df.shape[0]} invalid records")
    print(f" - Final sample size: {df.shape[0]:,}")
    print(f" - Control group size: {df[df['group'] == 'control'].shape[0]:,}")
    print(f" - Treatment group size: {df[df['group'] == 'treatment'].shape[0]:,}")
    
    return df

def hypothesis_testing(control: pd.Series, treatment: pd.Series, alpha: float = ALPHA) -> dict:
    """Execute full statistical hypothesis testing workflow"""
    from statsmodels.stats.power import NormalIndPower
    
    results = {}
    
    # Basic metrics
    n_control = len(control)
    n_treatment = len(treatment)
    cr_control = control.mean()
    cr_treatment = treatment.mean()
    abs_diff = cr_treatment - cr_control
    rel_diff = abs_diff / cr_control if cr_control != 0 else 0
    
    # Z-test for proportions
    z_stat, p_value = sms.proportions_ztest(
        [treatment.sum(), control.sum()],
        [n_treatment, n_control]
    )
    
    # Effect size calculation
    cohen_h = sms.proportion_effectsize(cr_control, cr_treatment)
    
    # Power analysis
    power_analyzer = NormalIndPower()
    power = power_analyzer.solve_power(
        effect_size=cohen_h,
        nobs1=n_treatment,
        alpha=alpha,
        ratio=n_control/n_treatment
    )
    
    # Confidence intervals
    conf_int = sms.confint_proportions_2indep(
        treatment.sum(), n_treatment,
        control.sum(), n_control,
        method='wald'
    )
    
    return {
        'control_rate': cr_control,
        'treatment_rate': cr_treatment,
        'absolute_diff': abs_diff,
        'relative_diff': rel_diff,
        'z_score': z_stat,
        'p_value': p_value,
        'cohen_h': cohen_h,
        'statistical_power': power,
        'confidence_interval': conf_int,
        'n_control': n_control,
        'n_treatment': n_treatment
    }

def generate_report(results: dict) -> None:
    """Generate comprehensive statistical reporting"""
    print("\n Final Statistical Report:")
    print(f"Control Conversion Rate: {results['control_rate']:.4f}")
    print(f"Treatment Conversion Rate: {results['treatment_rate']:.4f}")
    print(f"Absolute Difference: {results['absolute_diff']:.4f}")
    print(f"Relative Difference: {results['relative_diff']:.2%}")
    print(f"\nZ-score: {results['z_score']:.2f}")
    print(f"P-value: {results['p_value']:.4f}")
    print(f"95% Confidence Interval: ({results['confidence_interval'][0]:.4f}, {results['confidence_interval'][1]:.4f})")
    print(f"Cohen's h: {results['cohen_h']:.4f}")
    print(f"Statistical Power: {results['statistical_power']:.2%}")
    
    if results['p_value'] < ALPHA:
        if abs(results['absolute_diff']) > MIN_EFFECT_SIZE:
            print("\n Conclusion: Statistically significant and business meaningful difference detected")
        else:
            print("\n  Warning: Statistically significant but not business meaningful")
    else:
        print("\n Conclusion: No statistically significant difference detected")

def full_ab_test_analysis():
    """Complete A/B testing analysis workflow"""
    # Phase 1: Data Loading
    raw_df = load_and_validate_data(DATA_PATH)
    
    # Phase 2: Data Preparation
    clean_df = preprocess_data(raw_df)
    
    # Phase 3: Exploratory Analysis
    perform_eda(clean_df)
    
    # Phase 4: Statistical Testing
    control = clean_df[clean_df['group'] == 'control']['converted']
    treatment = clean_df[clean_df['group'] == 'treatment']['converted']
    
    test_results = hypothesis_testing(control, treatment)
    
    # Phase 5: Reporting
    generate_report(test_results)
    
    # Generate HTML report
    print("\n Generating detailed analysis report...")
    profile = ProfileReport(clean_df, title="A/B Testing Analysis Report")
    profile.to_file("ab_test_report.html")
    print(" Report saved as 'ab_test_report.html'")

if __name__ == "__main__":
    full_ab_test_analysis()
